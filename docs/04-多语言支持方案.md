### 4. 语言切换组件

```tsx
// components/LanguageSwitcher.tsx
'use client';

import { useRouter, usePathname } from 'next/navigation';
import { useLocale } from 'next-intl';

const LANGUAGES = [
  { code: 'en', name: 'English', native: 'English' },
  { code: 'zh-CN', name: '简体中文', native: '简体中文' },
  { code: 'de', name: 'German', native: 'Deutsch' },
  { code: 'fr', name: 'French', native: 'Français' },
  { code: 'es', name: 'Spanish', native: 'Español' },
  { code: 'ja', name: 'Japanese', native: '日本語' },
];

export function LanguageSwitcher() {
  const router = useRouter();
  const pathname = usePathname();
  const currentLocale = useLocale();

  const handleChange = (newLocale: string) => {
    // 保持当前路径，只改变语言前缀
    const segments = pathname.split('/');
    segments[1] = newLocale;
    const newPath = segments.join('/');
    router.push(newPath);
    
    // 保存用户偏好
    document.cookie = `NEXT_LOCALE=${newLocale}; path=/; max-age=31536000`;
  };

  return (
    <div className="relative">
      <select
        value={currentLocale}
        onChange={(e) => handleChange(e.target.value)}
        className="appearance-none bg-transparent border border-gray-300 rounded-md px-3 py-1.5 text-sm 
                   focus:outline-none focus:ring-2 focus:ring-indigo-500 cursor-pointer
                   hover:border-gray-400 transition-colors"
        aria-label="Select language"
      >
        {LANGUAGES.map((lang) => (
          <option key={lang.code} value={lang.code}>
            {lang.native}
          </option>
        ))}
      </select>
      
      {/* 下拉箭头图标 */}
      <div className="pointer-events-none absolute inset-y-0 right-0 flex items-center px-2">
        <svg className="h-4 w-4 text-gray-400" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 9l-7 7-7-7" />
        </svg>
      </div>
    </div>
  );
}
```

### 5. AI 翻译脚本

```typescript
// scripts/translate-articles.ts
import { writeFile, mkdir } from 'fs/promises';
import { generateObject } from 'ai';
import { createAnthropic } from '@ai-sdk/anthropic';

const anthropic = createAnthropic({ apiKey: process.env.ANTHROPIC_API_KEY });

interface TranslationResult {
  title: string;
  description: string;
  content: string;
  meta_title: string;
  meta_description: string;
}

async function translateArticle(
  sourceText: {
    title: string;
    description: string;
    content: string;
    meta_title: string;
    meta_description: string;
  },
  targetLang: string
): Promise<TranslationResult> {
  const prompt = `
    Translate the following technical article content to ${targetLang}.
    
    Guidelines:
    - Keep technical terms in English if no common translation exists
    - Keep code snippets, variable names, and URLs unchanged
    - Maintain the original tone (professional but friendly)
    - Keep markdown formatting intact
    - Translate meta descriptions for SEO
    
    Source content:
    ${JSON.stringify(sourceText, null, 2)}
  `;

  const { object } = await generateObject({
    model: anthropic('claude-sonnet-4'),
    schema: TranslationResult,
    prompt,
  });

  return object;
}

async function main() {
  // 读取英文文章
  const enArticles = await readJsonFile('messages/en/articles/*.json');
  
  for (const article of enArticles) {
    console.log(`Translating ${article.slug} to Chinese...`);
    
    const translation = await translateArticle(article, 'Simplified Chinese');
    
    // 保存翻译
    await mkdir('messages/zh-CN/articles', { recursive: true });
    await writeFile(
      `messages/zh-CN/articles/${article.slug}.json`,
      JSON.stringify(translation, null, 2)
    );
    
    console.log(`✓ Translated ${article.slug}`);
  }
}

main().catch(console.error);
```

### 6. 批量翻译 CLI 工具

```typescript
// scripts/translate-all.ts
#!/usr/bin/env ts-node
/**
 * 批量翻译工具
 * 用法：pnpm translate:all zh-CN
 */

import { glob } from 'glob';
import { readFile, writeFile, mkdir } from 'fs/promises';
import { generateText } from 'ai';
import { createAnthropic } from '@ai-sdk/anthropic';

const LANGUAGES = {
  'zh-CN': 'Simplified Chinese',
  'de': 'German',
  'fr': 'French',
  'es': 'Spanish',
  'ja': 'Japanese',
};

async function translateText(text: string, targetLang: string): Promise<string> {
  const { text: translated } = await generateText({
    model: createAnthropic()('claude-sonnet-4'),
    prompt: `Translate to ${targetLang}. Keep code unchanged. Maintain markdown format.
    
    Text:
    ${text}`,
  });
  
  return translated;
}

async function translateFile(inputPath: string, targetLang: string, outputPath: string) {
  const content = await readFile(inputPath, 'utf-8');
  const data = JSON.parse(content);
  
  const translated: Record<string, any> = {};
  
  for (const [key, value] of Object.entries(data)) {
    if (typeof value === 'string') {
      translated[key] = await translateText(value, targetLang);
    } else {
      translated[key] = value; // 保持非字符串不变
    }
  }
  
  await mkdir(path.dirname(outputPath), { recursive: true });
  await writeFile(outputPath, JSON.stringify(translated, null, 2));
}

async function main() {
  const [_, __, targetLang] = process.argv;
  
  if (!targetLang || !LANGUAGES[targetLang as keyof typeof LANGUAGES]) {
    console.error('Usage: pnpm translate:all <locale>');
    console.error('Available locales:', Object.keys(LANGUAGES));
    process.exit(1);
  }
  
  const langName = LANGUAGES[targetLang as keyof typeof LANGUAGES];
  console.log(`Starting translation to ${langName} (${targetLang})...`);
  
  // 翻译 UI 文本
  const uiFiles = await glob('messages/en/*.json');
  for (const file of uiFiles) {
    const fileName = path.basename(file);
    console.log(`Translating ${fileName}...`);
    await translateFile(
      file,
      langName,
      `messages/${targetLang}/${fileName}`
    );
  }
  
  // 翻译文章
  const articleFiles = await glob('messages/en/articles/*.json');
  for (const file of articleFiles) {
    const fileName = path.basename(file);
    console.log(`Translating article ${fileName}...`);
    await translateFile(
      file,
      langName,
      `messages/${targetLang}/articles/${fileName}`
    );
  }
  
  console.log('✓ Translation complete!');
}

main();
```

### 8. 自动语言检测

```typescript
// lib/detect-locale.ts
import { cookies, headers } from 'next/headers';

const SUPPORTED_LOCALES = ['en', 'zh-CN', 'de', 'fr', 'es', 'ja'];
const DEFAULT_LOCALE = 'en';

export function detectLocale(): string {
  // 1. 检查 Cookie（用户偏好）
  const cookieStore = cookies();
  const cookieLocale = cookieStore.get('NEXT_LOCALE')?.value;
  if (cookieLocale && SUPPORTED_LOCALES.includes(cookieLocale)) {
    return cookieLocale;
  }
  
  // 2. 检查 Accept-Language header
  const headersList = headers();
  const acceptLanguage = headersList.get('accept-language');
  
  if (acceptLanguage) {
    const preferred = parseAcceptLanguage(acceptLanguage);
    for (const lang of preferred) {
      // 精确匹配
      if (SUPPORTED_LOCALES.includes(lang)) {
        return lang;
      }
      
      // 语言前缀匹配（zh 匹配 zh-CN）
      const baseLang = lang.split('-')[0];
      const match = SUPPORTED_LOCALES.find(l => l.startsWith(baseLang));
      if (match) {
        return match;
      }
    }
  }
  
  // 3. 返回默认语言
  return DEFAULT_LOCALE;
}

function parseAcceptLanguage(header: string): string[] {
  return header
    .split(',')
    .map(lang => {
      const [locale, quality = 'q=1'] = lang.trim().split(';');
      return { locale, quality: parseFloat(quality.split('=')[1]) };
    })
    .sort((a, b) => b.quality - a.quality)
    .map(item => item.locale);
}
```

```typescript
    });
    
    // 文章页
    const articles = await getArticles(locale);
    for (const article of articles) {
      urls.push({
        url: `https://cyberpunkkb.com/${locale}/kb/${article.slug}`,
        lastModified: article.updatedAt,
        changeFrequency: 'weekly' as const,
        priority: 0.8,
      });
    }
  }
  
  return urls;
}
```

### 用户反馈机制

```tsx
// components/TranslationFeedback.tsx
'use client';

export function TranslationFeedback({ articleId, locale }: { articleId: string; locale: string }) {
  const [rating, setRating] = useState<number | null>(null);
  
  const submitFeedback = async (score: number) => {
    await fetch('/api/translation-feedback', {
      method: 'POST',
      body: JSON.stringify({ articleId, locale, score }),
    });
    setRating(score);
  };
  
  return (
    <div className="mt-8 p-4 bg-gray-50 rounded-lg">
      <p className="text-sm text-gray-600 mb-2">
        这个翻译质量如何？How is this translation?
      </p>
      <div className="flex gap-2">
        {[1, 2, 3, 4, 5].map((score) => (
          <button
            key={score}
            onClick={() => submitFeedback(score)}
            className={`w-8 h-8 rounded-full flex items-center justify-center
                       ${rating === score ? 'bg-indigo-600 text-white' : 'bg-gray-200 hover:bg-gray-300'}`}
          >
            {score}
          </button>
        ))}
      </div>
    </div>
  );
}
```
